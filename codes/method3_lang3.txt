import "math"

func JumpSearch(arr []int, target int) int {
    n := len(arr)
    step := int(math.Sqrt(float64(n)))
    prev := 0

    for arr[int(math.Min(float64(step), float64(n)))-1] < target {
        prev = step
        step += int(math.Sqrt(float64(n)))
        if prev >= n {
            return -1
        }
    }

    for i := prev; i < int(math.Min(float64(step), float64(n))); i++ {
        if arr[i] == target {
            return i
        }
    }

    return -1
}
